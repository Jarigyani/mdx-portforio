## 東京1F

### T-E-AS

```bash
! 初期設定
en
conf t
no ip domain lookup
hostname T-E-AS
username user01 password user01
enable secret enable

! vty
line vty 0 4
login local
exec-timeout 0 10
exit

! ssh
ip domain-name example.jp
crypto key generate rsa
2048
ip ssh version 2
line vty 0 4
transport input ssh
exit

! 管理IPアドレス設定
int vlan1
ip address 10.1.0.1 255.255.0.0
no shutdown
exit

! vlan作成
vlan 10
name Sales-T
exit

! スパニングツリー設定
spanning-tree mode rapid-pvst
spanning-tree vlan 10
spanning-tree vlan 10 priority 0

! ACL設定
access-list 1 permit 10.10.0.0 0.0.255.255
access-list 1 permit 10.100.0.0 0.0.255.255
access-list 1 permit 10.1.0.0 0.0.255.255

! インターフェース設定
int range f0/1-2
switchport mode access
switchport access vlan 10
ip access-group 1 in

! ポートセキュリティー設定
switchport port-security
switchport port-security 2
switchport port-security violation restrict
switchport port-security mac-address sticky
exit

! native vlan無いと管理IP接続できない
int range f0/23-24
switchport mode trunk
switchport trunk allowed vlan 1,10
switchport trunk native vlan 1
exit

! ntp設定
ntp server 10.10.10.3
ntp update-calendar

! 保存
end
wr
```

## 東京2F

### T-L3S1

```bash
! 初期設定
en
conf t
no ip domain lookup
service dhcp
ip routing
hostname T-L3S1
username user01 password user01
enable secret enable

! vty
line vty 0 4
login local
exec-timeout 0 10
exit

! ssh
ip domain-name example.jp
crypto key generate rsa
2048
ip ssh version 2
line vty 0 4
transport input ssh
exit

! 管理IPアドレス設定
int vlan1
ip address 10.1.0.6 255.255.0.0
no shutdown
exit

! vlan作成
vlan 10
name Sales-T
vlan 20
name Engineer
vlan 30
name General
exit

! インターフェース設定
int f0/1
switchport trunk encapsulation dot1q
switchport mode trunk
switchport trunk allowed vlan 1,10,20,30
switchport trunk native vlan 1
no shut
exit

int range f0/2-11
no shut
exit

int f0/23
no switchport
ip add 10.60.0.1 255.255.0.0
no shut

int f0/24
no switchport
ip add 10.80.0.1 255.255.0.0
no shut

int vlan 10
ip address 10.10.255.252 255.255.0.0
standby 10 ip 10.10.255.254
standby 10 priority 105
standby 10 preempt
standby 10 track FastEthernet0/23
standby 10 track FastEthernet0/24
exit

int vlan 20
ip address 10.20.255.252 255.255.0.0
standby 20 ip 10.20.255.254
standby 20 priority 105
standby 20 preempt
standby 20 track FastEthernet0/23
standby 20 track FastEthernet0/24
exit

int vlan 30
ip address 10.30.255.252 255.255.0.0
standby 30 ip 10.30.255.254
standby 30 priority 105
standby 30 preempt
standby 30 track FastEthernet0/23
standby 30 track FastEthernet0/24
exit

! DHCP設定
ip dhcp excluded-address 10.10.255.252 10.10.255.254

ip dhcp excluded-address 10.20.255.252 10.20.255.254

ip dhcp excluded-address 10.30.255.252 10.30.255.254

ip dhcp excluded-address 10.50.255.252 10.50.255.254

ip dhcp pool V10
network 10.10.0.0 255.255.0.0
default-router 10.10.255.254
exit

ip dhcp pool V20
network 10.20.0.0 255.255.0.0
default-router 10.20.255.254
exit

ip dhcp pool V30
network 10.30.0.0 255.255.0.0
default-router 10.30.255.254
exit

ip dhcp pool V50
network 10.50.0.0 255.255.0.0
default-router 10.50.255.254
exit

! OSPF設定
router ospf 1
network 10.10.0.0 0.0.255.255 area 0
network 10.20.0.0 0.0.255.255 area 0
network 10.30.0.0 0.0.255.255 area 0
network 10.60.0.0 0.0.255.255 area 0
network 10.80.0.0 0.0.255.255 area 0
exit

! 保存
end
wr
```
